''' Outputs average of list of positive integers List ends with 0 (sentinel) Ex: 10 1 6 3 0 yields (10 + 1 + 6 + 3) / 4, or 5 ''' values_sum = 0 num_values = 0 curr_value = float(input("Enter a positive integer (0 to quit):")) while curr_value > 0: # Get values until 0 (or less) values_sum += curr_value num_values += 1 curr_value = float(input()) print('Average: {}\n'.format(values_sum / num_values))
